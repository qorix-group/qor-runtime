10use core::types::{box_future, FutureBox};
11use std::future::Future;
12use std::sync::Arc;
13
14use scheduler::{
15    context::{ctx_get_handler, ctx_get_worker_id},
16    join_handle::JoinHandle,
17    task::async_task::{AsyncTask, TaskRef},
18};
19
20pub mod core;
21pub mod runtime;
22pub mod scheduler;
23
24///
25/// Spawns a given future into runtime and let it execute on any of configured workers
26///
27pub fn spawn<T>(future: T) -> JoinHandle<T::Output>
28where
29    T: Future + 'static + Send,
30    T::Output: Send,
31{
32    let boxed = box_future(future);
33    spawn_from_boxed(boxed)
34}
35
36///
37/// Same as `spawn` but from already boxed future
38///
39pub fn spawn_from_boxed<T>(boxed: FutureBox<T>) -> JoinHandle<T>
40where
41    T: 'static,
42{
43    if let Some(handler) = ctx_get_handler() {
44        let task = Arc::new(AsyncTask::new(boxed, ctx_get_worker_id(), handler.scheduler.clone()));
45        handler.spawn(task)
46    } else {
47        panic!("For now we don't allow runtime API to be called outside of runtime")
48    }
49}
