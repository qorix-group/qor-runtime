28static EVENT_OBJ: LazyLock<Mutex<Event>> = LazyLock::new(|| {
29    let mut evts: HashMap<usize, (Listener<ipc::Service>, Option<Waker>, bool)> = HashMap::new();
30    // The internal event name shall be unique within the system. Otherwise, when two or more processes running, the trigger will be delivered to all.
31    let internal_event_name = "qorix_internal_waker_".to_string() + &process::id().to_string();
32
33    let config = {
34        let mut config = Config::default();
35        config.global.prefix = "orch_node".try_into().unwrap();
36        config
37    };
38
39    Node::<ipc::Service>::cleanup_dead_nodes(&config);
40    Node::<ipc::Service>::list(&config, |node_state| {
41        if let NodeState::<ipc::Service>::Dead(view) = node_state {
42            if let Err(e) = view.remove_stale_resources() {
43                error!("Failed to clean iceoryx2 resources: {:?}", e);
44            }
45        }
46        CallbackProgression::Continue
47    })
48    .expect("failed clenup node stall state");
49
50    let name = NodeName::new(&format!("orch_node_{}", process::id())).expect("Broken node name");
51    let node = NodeBuilder::new().name(&name).config(&config).create::<ipc::Service>().unwrap();
52
53    let internal_notifier = node
54        .service_builder(&internal_event_name.as_str().try_into().unwrap())
55        .event()
56        .open_or_create()
57        .unwrap()
58        .notifier_builder()
59        .create()
60        .unwrap();
61
62    let internal_listener = node
63        .service_builder(&internal_event_name.as_str().try_into().unwrap())
64        .event()
65        .open_or_create()
66        .unwrap()
67        .listener_builder()
68        .create()
69        .unwrap();
70
71    evts.insert(0, (internal_listener, None, false));
72
73    Mutex::new(Event {
74        service_node: node,
75        events: evts,
76        internal_notifier: internal_notifier,
77        notifiers: HashMap::new(),
78    })
79});

81pub struct Event {
82    service_node: Node<ipc::Service>,
83    // Hash Map: event name (for debugging/logging), waker, bool flag to indicate whether event was received already. It is updated by polling thread.
84    events: HashMap<usize, (Listener<ipc::Service>, Option<Waker>, bool)>,
85    internal_notifier: Notifier<ipc::Service>,
86    notifiers: HashMap<String, Notifier<ipc::Service>>,
87}
88
89/// Singleton Event
90impl Event {
91    /// To get the instance of Event
92    pub fn get_instance() -> &'static Mutex<Event> {
93        &EVENT_OBJ
94    }
95
96    /// To be called when the trigger action is created as part of program
97    pub fn create_notifier(&mut self, event_name: &str) {
98        let event = self
99            .service_node
100            .service_builder(&event_name.try_into().unwrap())
101            .event()
102            .open_or_create()
103            .unwrap();
104
105        let notifier = event.notifier_builder().create().unwrap();
106        self.notifiers.entry(event_name.to_string()).or_insert(notifier);
107    }
108
109    /// To be called when trigger action is executed
110    pub fn trigger_event(&self, event_name: &str) -> ActionResult {
111        match self.notifiers.get(event_name).unwrap().notify() {
112            Ok(_) => Ok(()),
113            _ => Err(CommonErrors::GenericError),
114        }
115    }
116
117    /// To be called when the sync action is created as part of program
118    /// This stores the event name. The listeners will be created in polling thread to create waitset & wait for all listeners.
119    pub fn create_listener(&mut self, event_name: &str) -> usize {
120        let event = self
121            .service_node
122            .service_builder(&event_name.try_into().unwrap())
123            .event()
124            .open_or_create()
125            .unwrap();
126
127        let listener = event.listener_builder().create().unwrap();
128
129        // The event ID/key starts from 0 to n-1
130        let event_id = self.events.len();
131        self.events.insert(event_id, (listener, None, false));
132
133        self.internal_notifier.notify().unwrap(); // Will panic if there is error
134
135        // Return key as ID of the listener.
136        return event_id;
137    }
138
139    /// This is to check whether the event is received or not
140    pub fn check_event(&mut self, event_id: usize) -> bool {
141        // Check the event received 'flag' updated by polling thread.
142        if let Some((_, _, event_received)) = self.events.get_mut(&event_id) {
143            if *event_received == true {
144                *event_received = false;
145                debug!("Received event: {}", event_id);
146                return true;
147            }
148        }
149        return false;
150    }
151
152    /// To be called when sync action is executed
153    pub fn wake_on_event(&mut self, event_id: usize, waker_in: Waker) -> bool {
154        if self.check_event(event_id) == false {
155            if let Some((_, waker, _)) = self.events.get_mut(&event_id) {
156                *waker = Some(waker_in);
157                debug!("Adding waker for event: {}", event_id);
158            }
159            // Event was not received
160            return false;
161        } else {
162            // Event already received.
163            return true;
164        }
165    }
166
167    pub fn create_polling_thread(&self) {
168        let event_obj = Event::get_instance();
169
170        thread::spawn(move || {
171            // Set the priority of Event Handler Thread to appropriate higher value for faster response to event.
172            // The executable shall have CAP_SYS_NICE capability on Linux / PROCMGR_AID_PRIORITY capability on Qnx for priorities above 63.
173            // The CAP_SYS_NICE capability can be enabled by executing "sudo setcap cap_sys_nice=eip <executable_name>" on terminal.
174            unsafe {
175                let param = sched_param { sched_priority: 50 }; // Priority shall be configurable between 1-99
176                let pid = libc::gettid(); // Get thread ID
177                let result = sched_setscheduler(pid, SCHED_FIFO, &param); // Policy shall be configurable
178                if result == 0 {
179                    debug!("[EHT] Thread priority set successfully");
180                } else {
181                    debug!("[EHT] Failed to set priority");
182                }
183            }
184            debug!("[EHT] Event handler thread running...");
185
186            'outer: loop {
187                let mut event_id_and_listener_fd = vec![]; // (event_id, listener fd)
188                let mut poll_fds = vec![];
189                {
190                    let events_map = &event_obj.lock().unwrap().events;
191                    for (event_id, (listener, _, _)) in events_map {
192                        let listener_fd = unsafe { listener.file_descriptor().native_handle() };
193                        event_id_and_listener_fd.push((*event_id, listener_fd));
194
195                        // Create pollfd for each listener fd
196                        let poll_fd = pollfd {
197                            fd: listener_fd,
198                            events: POLLIN,
199                            revents: 0,
200                        };
201                        poll_fds.push(poll_fd);
202                    }
203                }
204
205                let mut update_poll_fds = false;
206                // loops until the user has pressed CTRL+c, the application has received a SIGTERM or SIGINT
207                loop {
208                    // Blocked poll(), no timeout, return if event or error detected.
209                    let result = unsafe { poll(poll_fds.as_mut_ptr(), poll_fds.len() as libc::nfds_t, -1) };
210
211                    if result > 0 {
212                        // Some events received, process it.
213                        for index in 0..poll_fds.len() {
214                            if poll_fds[index].revents & POLLIN == POLLIN {
215                                // Read the data and empty it.
216                                let mut buf = [0u8; 128];
217                                let recv_size = unsafe { libc::recv(poll_fds[index].fd, buf.as_mut_ptr() as *mut _, buf.len(), 0) };
218
219                                if recv_size > 0 {
220                                    if let Some((event_id, _)) = event_id_and_listener_fd.get(index) {
221                                        if *event_id == 0 {
222                                            // Internal event. Just set the flag and process others.
223                                            update_poll_fds = true;
224                                        } else {
225                                            if let Some((_, waker, event_received)) = event_obj.lock().unwrap().events.get_mut(event_id) {
226                                                trace!("[EHT] Received event: {}", event_id);
227
228                                                // Set the flag for not to miss any event. This flag can be checked in wake_on_event function.
229                                                *event_received = true;
230
231                                                // If there is any waker configured i.e.sync action is waiting for event, wake it and remove waker.
232                                                if let Some(waker_obj) = waker {
233                                                    trace!("[EHT] Wake awaiting thread using waker for event: {}", event_id);
234                                                    waker_obj.wake_by_ref();
235                                                    *waker = None;
236                                                }
237                                            }
238                                        }
239                                    }
240                                } else if recv_size == -1 {
241                                    // Error, just ignore now.
242                                    warn!("Read error, ignored.")
243                                }
244                            }
245                        }
246                    } else if result == -1 {
247                        // Error, exit thread.
248                        break 'outer;
249                    } else {
250                        // Timeout, this should not happen.
251                    }
252
253                    // If there are newly added listeners, then exit inner loop to update poll_fds.
254                    if update_poll_fds {
255                        break;
256                    }
257                }
258            }
259
260            debug!("[EHT] Exiting event handler thread.");
261        });
262        // Sleep to ensure tha event handler thread gets a chance to run and change its priority.
263        thread::sleep(Duration::from_millis(1));
264    }
265}
