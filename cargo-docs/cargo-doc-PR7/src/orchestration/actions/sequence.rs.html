10use async_runtime::core::types::box_future;
11use logging_tracing::prelude::*;
12
13use super::action::{ActionBaseMeta, ActionFuture, ActionResult, ActionTrait, NamedId};
14
15pub struct Sequence {
16    base: ActionBaseMeta,
17    actions: Vec<Box<dyn ActionTrait>>,
18}
19
20impl Sequence {
21    /// Create a new sequence action
22    pub fn new() -> Box<Sequence> {
23        Self::new_internal(NamedId::default())
24    }
25
26    pub fn new_with_id(id: NamedId) -> Box<Sequence> {
27        Self::new_internal(id)
28    }
29
30    fn new_internal(named_id: NamedId) -> Box<Sequence> {
31        Box::new(Self {
32            actions: Vec::new(),
33            base: ActionBaseMeta {
34                named_id,
35                runtime: Default::default(),
36            },
37        })
38    }
39
40    /// Add sequence step
41    pub fn with_step(mut self: Box<Self>, action: Box<dyn ActionTrait>) -> Box<Self> {
42        self.actions.push(action);
43        self
44    }
45
46    /// Execute a futures collection and terminates immediately upon error
47    async fn execute_impl(meta: ActionBaseMeta, mut futures: Vec<ActionFuture>) -> ActionResult {
48        let mut result = Ok(());
49
50        trace!(sequence = ?meta, "Before joining steps");
51
52        for future in futures.iter_mut() {
53            result = future.await;
54            if result.is_err() {
55                // terminate sequence and propagate the error
56                error!("error in step!");
57                break;
58            }
59        }
60
61        trace!(sequence = ?meta, "After joining steps");
62        result
63    }
64}
65
66impl ActionTrait for Sequence {
67    /// Will be called on each sequence step
68    fn execute(&mut self) -> ActionFuture {
69        // first collect all futures from the steps
70        let futures: std::vec::Vec<_> = self.actions.iter_mut().map(|action| action.execute()).collect();
71
72        // and execute them
73        box_future(Sequence::execute_impl(self.base, futures))
74    }
75
76    fn name(&self) -> &'static str {
77        "Sequence"
78    }
79
80    fn dbg_fmt(&self, nest: usize, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
81        let indent = " ".repeat(nest);
82        writeln!(f, "{}|-{} - {:?}", indent, self.name(), self.base)?;
83        self.actions.iter().try_for_each(|x| {
84            writeln!(f, "{} |step", indent)?;
85            x.dbg_fmt(nest + 1, f)
86        })
87    }
88
89    fn fill_runtime_info(&mut self, p: &mut super::action::ActionRuntimeInfoProvider) {
90        self.base.runtime = p.next();
91        self.actions.iter_mut().for_each(|item| item.fill_runtime_info(p));
92    }
93}
