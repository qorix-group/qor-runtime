10use super::task::async_task::*;
11use core::task::{RawWaker, RawWakerVTable, Waker};
12
13fn clone_waker(data: *const ()) -> RawWaker {
14    let task_header_ptr = data as *const TaskHeader;
15    let task_ref = unsafe { TaskRef::from_raw(task_header_ptr) };
16
17    let new_waker = task_ref.clone();
18
19    std::mem::forget(task_ref); // We need to make sure the instance from which we clone, is forgotten since we did not consumed it, it was only bring back for a moment to clone
20
21    let raw = TaskRef::into_raw(new_waker);
22    RawWaker::new(raw as *const (), &VTABLE)
23}
24
25fn wake(data: *const ()) {
26    let task_header_ptr = data as *const TaskHeader;
27    let task_ref = unsafe { TaskRef::from_raw(task_header_ptr) };
28
29    task_ref.schedule();
30
31    drop(task_ref); // wake uses move semantic, so we are owner of data now, so we need to cleanup
32}
33
34fn wake_by_ref(data: *const ()) {
35    let task_header_ptr = data as *const TaskHeader;
36    let task_ref = unsafe { TaskRef::from_raw(task_header_ptr) };
37
38    task_ref.schedule();
39
40    std::mem::forget(task_ref); // don't touch refcount from our data since this is done by drop_waker
41}
42
43fn drop_waker(data: *const ()) {
44    let task_header_ptr = data as *const TaskHeader;
45    let task_ref = unsafe { TaskRef::from_raw(task_header_ptr) };
46
47    drop(task_ref); // We get rid of instance, so ref count will be decremented
48}
49
50// Define the RawWakerVTable
51static VTABLE: RawWakerVTable = RawWakerVTable::new(clone_waker, wake, wake_by_ref, drop_waker);
52
53///
54/// Waker will store internally a pointer to the ref counted Task.
55///
56pub(crate) fn create_waker(ptr: TaskRef) -> Waker {
57    let ptr = TaskRef::into_raw(ptr); // Extracts the pointer from TaskRef not decreasing it's reference count. Since we have a clone here, ref cnt was already increased
58    let raw_waker = RawWaker::new(ptr as *const (), &VTABLE);
59
60    // Convert RawWaker to Waker
61    unsafe { Waker::from_raw(raw_waker) }
62}
