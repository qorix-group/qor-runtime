19pub type FutureBox<T: Send> = Pin<BoxCustom<dyn Future<Output = T> + Send>>;

29///
30/// TaskId encodes the worker on which it was created and it's number local to the worker.
31/// This id cannot be used to infer task order creation or anything like that, it's only for identification purpose.
32///
33#[derive(Copy, Clone, Debug)]
34pub(crate) struct TaskId(pub(crate) u32);
35
36thread_local! {
37    static TASK_COUNTER: Cell<u32> = Cell::new(0);
38}
39
40impl TaskId {
41    pub(crate) fn new(worker_id: u8) -> Self {
42        let val = (TASK_COUNTER.get()) % 0x00FFFFFF; //TODO: Fix it later or change algo
43        TASK_COUNTER.set(val + 1);
44
45        Self((val << 8) | worker_id as u32)
46    }
47
48    pub(crate) fn worker(&self) -> u8 {
49        (self.0 & 0xFF as u32) as u8
50    }
51}
