10use std::cell::UnsafeCell;
11use std::ops::{Deref, DerefMut};
12
13use foundation::prelude::IoxAtomicBool;
14
15pub struct RuntimeSequentialAccess<T> {
16    data: UnsafeCell<T>,
17    is_used: IoxAtomicBool,
18}
19
20// Safety: `RuntimeSequentialAccess` assumes single-threaded execution, so we manually implement `Send` and `Sync`
21// Attention: This is unfinished as it shall act more like async mutex which goes into sleep once there is no lock available
22unsafe impl<T> Send for RuntimeSequentialAccess<T> {}
23unsafe impl<T> Sync for RuntimeSequentialAccess<T> {}
24
25impl<T> RuntimeSequentialAccess<T> {
26    pub fn new(value: T) -> Self {
27        Self {
28            data: UnsafeCell::new(value),
29            is_used: IoxAtomicBool::new(false),
30        }
31    }
32
33    pub fn is_locked(&self) -> bool {
34        self.is_used.load(std::sync::atomic::Ordering::SeqCst)
35    }
36
37    pub fn lock(&self) -> RuntimeSequentialAccessGuard<T> {
38        if let Err(_) = self
39            .is_used
40            .compare_exchange(false, true, std::sync::atomic::Ordering::SeqCst, std::sync::atomic::Ordering::SeqCst)
41        {
42            panic!("Trying to take a fake lock in orchestration from owned object while this object is being executed");
43        }
44
45        RuntimeSequentialAccessGuard { fake_mtx: self }
46    }
47}
48
49// Scoped guard that allows mutable access while it's held
50pub struct RuntimeSequentialAccessGuard<'a, T> {
51    fake_mtx: &'a RuntimeSequentialAccess<T>,
52}
53
54impl<T> Drop for RuntimeSequentialAccessGuard<'_, T> {
55    fn drop(&mut self) {
56        self.fake_mtx.is_used.store(false, std::sync::atomic::Ordering::SeqCst);
57    }
58}
59
60impl<T> Deref for RuntimeSequentialAccessGuard<'_, T> {
61    type Target = T;
62
63    fn deref(&self) -> &T {
64        unsafe { &*self.fake_mtx.data.get() }
65    }
66}
67
68impl<T> DerefMut for RuntimeSequentialAccessGuard<'_, T> {
69    fn deref_mut(&mut self) -> &mut T {
70        unsafe { &mut *self.fake_mtx.data.get() }
71    }
72}
