fn init(&mut self) -> ExecutionState

Initialize a new action with the given actionable. This is a valid operation when the action is Uninitialized or Terminated.

Source

fn start(&mut self, ctx: &mut ExecutionContext<'_>) -> ExecutionState

Start the action. This is a valid operation after the action has been initialized Ready or stopped Completed.

Source

fn update( &mut self, delta: &Duration, ctx: &mut ExecutionContext<'_>, ) -> (Duration, UpdateResult)

Update the action with the given time delta. This is a valid operation after the action has been started and is Running.

Source

fn finalize(&mut self, ctx: &mut ExecutionContext<'_>) -> ExecutionState

Stop and finalize the action. This is a valid operation when the action is Running.

Source

fn terminate(&mut self) -> ExecutionState

Terminate the action. This is a valid operation when the action is Ready, Completed or in Err state.
