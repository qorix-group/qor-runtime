22pub struct Handler {
23    pub(crate) scheduler: Arc<Scheduler>,
24    prod_con: Rc<BoundProducerConsumer<TaskRef>>, // Worker queue producer-consumer object
25}
26
27///
28/// Implements proxy API between Runtime API and {Scheduler, ...}
29///
30impl Handler {
31    pub(crate) fn spawn<T>(&self, task: Arc<AsyncTask<T>>) -> JoinHandle<T> {
32        let handle = JoinHandle::new(task.clone());
33        let task_ref = TaskRef::new(task.clone());
34
35        self.scheduler.spawn_from_runtime(task_ref, &self.prod_con);
36        handle
37    }
38
39    pub(super) fn respawn(&self, task_ref: TaskRef) {
40        self.scheduler.spawn_from_runtime(task_ref, &self.prod_con);
41    }
42}

44///
45/// This is an entry point for public API that is filled by each worker once it's created
46///
47pub(crate) struct WorkerContext {
48    /// OS Thread ID that the current context was bound too
49    thread_id: u64,
50
51    /// The ID of task that is currently run by worker
52    running_task_id: Cell<Option<TaskId>>,
53
54    /// WorkerID and EngineID
55    worker_type: Cell<WorkerType>,
56
57    /// Access to scheduler and others, mounted in pre_run phase of each Worker
58    pub(super) handler: RefCell<Option<Rc<Handler>>>,
59}
60
61thread_local! {
62    static CTX: RefCell<Option<WorkerContext>> = RefCell::new(None)
63}
64
65pub(crate) struct ContextBuidler {
66    tid: u64,
67    handle: Option<Handler>,
68    worker_type: Option<WorkerType>,
69}
70
71impl ContextBuidler {
72    pub(crate) fn new() -> Self {
73        Self {
74            tid: 0,
75            handle: None,
76            worker_type: None,
77        }
78    }
79
80    pub(crate) fn thread_id(mut self, id: u64) -> Self {
81        self.tid = id;
82        self
83    }
84
85    pub(crate) fn with_worker_type(mut self, t: WorkerType) -> Self {
86        self.worker_type = Some(t);
87        self
88    }
89
90    pub(crate) fn with_handle(mut self, pc: Rc<BoundProducerConsumer<TaskRef>>, s: Arc<Scheduler>) -> Self {
91        let handle = Handler { prod_con: pc, scheduler: s };
92
93        self.handle = Some(handle);
94        self
95    }
96
97    pub(crate) fn build(self) -> WorkerContext {
98        WorkerContext {
99            thread_id: self.tid,
100            running_task_id: Cell::new(None),
101            worker_type: Cell::new(self.worker_type.expect("Worker type must be set in context builder!")),
102            handler: RefCell::new(Some(Rc::new(self.handle.expect("Handler type must be set in context builder!")))),
103        }
104    }
105}

107///
108/// Needs to be called at worker thread init, so runtime API is functional from worker context
109///
110pub(crate) fn ctx_initialize(builder: ContextBuidler) {
111    let _ = CTX
112        .try_with(|ctx| {
113            let prev = ctx.replace(Some(builder.build()));
114            if prev.is_some() {
115                panic!("Double init of WorkerContext is not allowed!");
116            }
117        })
118        .map_err(|e| {
119            panic!("Something is really bad here, error {}!", e);
120        });
121}
122
123///
124/// Returns `Handler` for scheduler or None if not in context.
125///
126pub(crate) fn ctx_get_handler() -> Option<Rc<Handler>> {
127    CTX.try_with(|ctx| ctx.borrow().as_ref()?.handler.borrow().as_ref().cloned())
128        .unwrap_or(None)
129}
130
131///
132/// Sets currently running `task id`
133///
134pub(crate) fn ctx_set_running_task_id(id: TaskId) {
135    let _ = CTX
136        .try_with(|ctx| {
137            ctx.borrow().as_ref().expect("Called before CTX init?").running_task_id.replace(Some(id));
138        })
139        .map_err(|e| {
140            panic!("Something is really bad here, error {}!", e);
141        });
142}
143
144///
145/// Clears currently running `task id`
146///
147pub(crate) fn ctx_unset_running_task_id() {
148    let _ = CTX
149        .try_with(|ctx| {
150            ctx.borrow().as_ref().expect("Called before CTX init?").running_task_id.replace(None);
151        })
152        .map_err(|e| {});
153}
154
155///
156/// Gets currently running `task id`
157///
158pub(crate) fn ctx_get_running_task_id() -> Option<TaskId> {
159    CTX.try_with(|ctx| ctx.borrow().as_ref().expect("Called before CTX init?").running_task_id.get())
160        .unwrap_or_else(|e| {
161            panic!("Something is really bad here, error {}!", e);
162        })
163}
164
165///
166/// Worker id bound to this context
167///
168pub(crate) fn ctx_get_worker_id() -> u8 {
169    CTX.try_with(|ctx| {
170        let handler = ctx.borrow();
171        let b = handler.as_ref().expect("Called before CTX init?");
172
173        let typ = b.worker_type.get();
174        match typ {
175            WorkerType::EngineId((_, id)) => id,
176            WorkerType::DedicatedWorkerId(_) => todo!(),
177            WorkerType::None => todo!(),
178        }
179    })
180    .unwrap_or_else(|e| {
181        panic!("Something is really bad here, error {}!", e);
182    })
183}
