17type ActionMeta = (Option<ActionFuture>, Option<JoinHandle<ActionResult>>);
18type ActionsVectorType = Vec<ActionMeta>;
19
20///
21/// Branches execution flow into separate, independent paths using a [`async_runtime::spawn`] which will issue an concurrent execution depending on runtime workers configuration
22///
23pub struct Concurrency {
24    base: ActionBaseMeta,
25    actions: Vec<Box<dyn ActionTrait>>,
26}
27
28///
29/// Internal future that can wait for multiple [`JoinHandle`] and return Ready once all are done or any fails canceling all other handles.
30///
31struct ConcurrencyJoin<'a> {
32    handles: &'a mut [ActionMeta],
33}
34
35impl<'a> ConcurrencyJoin<'a> {
36    fn new(actions: &'a mut ActionsVectorType) -> Self {
37        Self { handles: actions }
38    }
39}
40
41impl Future for ConcurrencyJoin<'_> {
42    type Output = ActionResult;
43
44    fn poll(mut self: std::pin::Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> std::task::Poll<Self::Output> {
45        let mut is_done = true;
46
47        for (_, handle_opt) in &mut *self.handles {
48            match handle_opt {
49                Some(handle) => {
50                    let p = Pin::new(handle);
51                    let res = p.poll(cx);
52
53                    match res {
54                        std::task::Poll::Ready(_) => {
55                            *handle_opt = None; // Clear out handle that is already done
56
57                            // TODO: Missing abort logic & error propagation for now
58                        }
59                        std::task::Poll::Pending => {
60                            is_done = false;
61                        }
62                    }
63                }
64                None => {
65                    //Nothing to do, this handle was already consumed
66                }
67            }
68        }
69
70        if is_done {
71            std::task::Poll::Ready(Ok(()))
72        } else {
73            std::task::Poll::Pending
74        }
75    }
76}

78impl Concurrency {
79    pub fn new() -> Box<Concurrency> {
80        Self::new_internal(NamedId::default())
81    }
82
83    pub fn new_with_id(id: NamedId) -> Box<Concurrency> {
84        Self::new_internal(id)
85    }
86
87    fn new_internal(named_id: NamedId) -> Box<Concurrency> {
88        Box::new(Self {
89            actions: Vec::new(),
90            base: ActionBaseMeta {
91                named_id,
92                runtime: Default::default(),
93            },
94        })
95    }
96
97    ///
98    /// Adds new branch to flow execution
99    ///
100    pub fn with_branch(mut self: Box<Self>, action: Box<dyn ActionTrait>) -> Box<Self> {
101        self.actions.push(action);
102        self
103    }
104
105    //
106    // PRIVATE
107    //
108
109    async fn internal_future(meta: ActionBaseMeta, mut futures_collection: ActionsVectorType) -> ActionResult {
110        for (fut_opt, handle_opt) in &mut futures_collection {
111            match fut_opt.take() {
112                Some(fut) => {
113                    *handle_opt = Some(async_runtime::spawn_from_boxed(fut));
114                }
115                None => {
116                    panic!("We shall never be here since we always create Some(fut) at beginning")
117                }
118            }
119        }
120
121        trace!(concurrent = ?meta, "Before joining branches");
122
123        let joined = ConcurrencyJoin::new(&mut futures_collection);
124        let res = joined.await;
125
126        trace!(concurrent = ?meta, "After joining branches");
127        res
128    }
129}
130
131impl ActionTrait for Concurrency {
132    fn execute(&mut self) -> ActionFuture {
133        //TODO: Here we have allocation, discuss how we can avoid it, but it does not influence a TaskFlow, it only influence each run of it (since allocation is done before each program iteration)
134        let collected_futures: ActionsVectorType = self.actions.iter_mut().map(|action| (Some(action.execute()), None)).collect();
135
136        box_future(Self::internal_future(self.base, collected_futures))
137    }
138
139    fn name(&self) -> &'static str {
140        "Concurrency"
141    }
142
143    fn dbg_fmt(&self, nest: usize, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
144        let indent = " ".repeat(nest);
145        writeln!(f, "{}|-{} - {:?}", indent, self.name(), self.base)?;
146        self.actions.iter().try_for_each(|x| {
147            writeln!(f, "{} |branch", indent)?;
148            x.dbg_fmt(nest + 1, f)
149        })
150    }
151
152    fn fill_runtime_info(&mut self, p: &mut super::action::ActionRuntimeInfoProvider) {
153        self.base.runtime = p.next();
154        self.actions.iter_mut().for_each(|item| item.fill_runtime_info(p));
155    }
156}
