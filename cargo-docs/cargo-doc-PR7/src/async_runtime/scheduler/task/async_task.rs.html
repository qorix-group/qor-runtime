18///
19/// Table of pointers to access API of generic `AsyncTask` without a need to know it's type along a way
20///
21struct TaskVTable {
22    ///
23    /// Drops one ref count from associated ArcInternal<AsyncTask<T>>
24    ///
25    drop: unsafe fn(NonNull<TaskHeader>),
26
27    ///
28    /// Adds one ref count from associated ArcInternal<AsyncTask<T>>
29    ///
30    clone: unsafe fn(NonNull<TaskHeader>),
31
32    ///
33    /// Reschedules task into runtime
34    ///
35    schedule: unsafe fn(NonNull<TaskHeader>, TaskRef),
36
37    ///
38    ///
39    ///
40    poll: unsafe fn(this: NonNull<TaskHeader>, ctx: &mut Context) -> TaskPollResult,
41}

43///
44/// Task stage in which we are in
45///
46pub(crate) enum TaskStage<T, ResultType> {
47    Extracted,             // No data available anymore
48    InProgress(T),         // Holds "a thing" to execute (could be future, could be fn)
49    Completed(ResultType), // Future finished and return value is stored in here
50}

77///
78/// Represents async task with a future. Keep in mind that this can only be accessed via `Task`` trait after construction
79///
80/// Safety:
81///  - has to be repr(C) to make TaskRef work
82///
83/// Design decision
84///    We took a decision for `T` being a `Future Return Type`` and not the Future itself. This is because each Future has different size and `AsyncTask` need to somehow dynamically allocated. Assuming
85///    Future is Boxed, `AsyncTask` will have predictable sizes which are easier to map into some simple allocators like MemPool. If we some day device to abandon this, there will be not so much change
86///    around here or we can even implement new Task class out of current building blocks and it will fit to whole system seamlessly.
87#[repr(C)]
88pub(crate) struct AsyncTask<T: 'static> {
89    pub(in crate::scheduler) header: TaskHeader,
90    stage: UnsafeCell<TaskStage<FutureBox<T>, T>>, // Describe which stage we are in (progress, done, etc)
91
92    handle_waker: UnsafeCell<Option<Waker>>, // Waker for the one that hold the JoinHandle, for now Option, but potentially needs a wrapper for sharing too
93    scheduler: ArcInternal<Scheduler>,
94}
95
96// We protect *Cells by task state, making sure there is no concurrency on those
97unsafe impl<T: 'static> Sync for AsyncTask<T> {}
98
99impl<T> AsyncTask<T> {
100    pub(crate) fn new(future: FutureBox<T>, worker_id: u8, scheduler: ArcInternal<Scheduler>) -> Self {
101        Self {
102            header: TaskHeader::new::<T>(worker_id),
103            stage: UnsafeCell::new(TaskStage::InProgress(future)),
104            handle_waker: UnsafeCell::new(None),
105            scheduler,
106        }
107    }
108
109    pub(crate) fn id(&self) -> TaskId {
110        self.header.id
111    }
112
113    pub(crate) fn set_waker(&self, waker: Waker) {
114        unsafe {
115            *(self.handle_waker.get()) = (Some(waker));
116        }
117    }
118
119    pub(crate) fn poll(&self, cx: &mut Context) -> TaskPollResult {
120        match self.header.state.transition_to_running() {
121            TransitionToRunning::Done => {
122                // means we are the only one who could be now running this future
123                self.poll_core(cx)
124            }
125            TransitionToRunning::Aborted => {
126                // TODO: Add abort handler, but this mean we are basically done for outer level, for now marked as todo!() for final impl
127                todo!();
128            }
129        }
130    }
131
132    ///
133    /// Safety: outer caller needs to ensure that the stage is synchronized by task state
134    ///
135    fn poll_core(&self, ctx: &mut Context) -> TaskPollResult {
136        // Once we are here, we are the only one who can be running this future and/or stage itself
137
138        let ref_to_stage = unsafe { &mut *self.stage.get() };
139
140        match ref_to_stage {
141            TaskStage::Extracted => {
142                panic!("We shall never be in TaskStage::Extracted in poll as we handle that before (task state == completed)")
143            }
144            TaskStage::InProgress(ref mut future) => {
145                // Lets poll future
146                match future.as_mut().poll(ctx) {
147                    std::task::Poll::Ready(ret) => {
148                        // Store result
149                        *ref_to_stage = TaskStage::Completed(ret);
150
151                        let status = self.header.state.transition_to_completed();
152                        match status {
153                            TransitionToCompleted::Done => {} // No join handle connected yet, means join handle will make sure its not sleeping on its own on first poll
154                            TransitionToCompleted::HadConnectedJoinHandle => match unsafe { &*self.handle_waker.get() } {
155                                Some(v) => {
156                                    v.wake_by_ref();
157                                }
158                                None => panic!("We shall never be here if we have HadConnectedJoinHandle set!"),
159                            },
160                        }
161
162                        TaskPollResult::Done
163                    }
164                    std::task::Poll::Pending => {
165                        match self.header.state.transition_to_idle() {
166                            TransitionToIdle::Done => TaskPollResult::Done,
167                            TransitionToIdle::Notified => TaskPollResult::Notified,
168                            TransitionToIdle::Aborted => {
169                                // TODO: Add abort handler, but this mean we are basically done for outer level, for now marked as todo!() for final impl
170                                todo!();
171                            }
172                        }
173                    }
174                }
175            }
176            TaskStage::Completed(_) => {
177                panic!("We shall never be in TaskStage::Completed in poll as we handle that before (task state == completed)")
178            }
179        }
180    }
181
182    pub fn get_future_ret(&self) -> Result<T, CommonErrors> {
183        if !self.header.state.is_completed() {
184            return Err(CommonErrors::NoData);
185        }
186
187        let ref_to_stage = unsafe { &mut *self.stage.get() };
188
189        let should_take = match ref_to_stage {
190            TaskStage::Extracted => false,
191            TaskStage::InProgress(_) => false,
192            TaskStage::Completed(_) => true,
193        };
194
195        let prev = mem::replace(&mut *ref_to_stage, TaskStage::Extracted);
196
197        if should_take {
198            if let TaskStage::Completed(v) = prev {
199                Ok(v)
200            } else {
201                Err(CommonErrors::AlreadyDone)
202            }
203        } else {
204            Err(CommonErrors::AlreadyDone)
205        }
206    }
207
208    fn schedule(&self, task: TaskRef) {
209        // TODO: For now we do simple reschedule (even not in instance, for now not needed, but later will be needed) without:
210        // - no spawn into certain place (certain worker)
211        // - ...
212
213        match self.header.state.set_notified() {
214            TransitionToNotified::Done => {
215                // We need to reschedule on our own
216
217                if let Some(handler) = ctx_get_handler() {
218                    handler.respawn(task);
219                } else {
220                    self.scheduler.spawn_outside_runtime(task);
221                }
222            }
223            TransitionToNotified::AlreadyNotified => {
224                // Do nothing as someone already did it
225            }
226            TransitionToNotified::Running => {
227                // Do nothing was we will be rescheduled by poll itself, still notification was marked
228            }
229        }
230    }
231
232    /// instance unbounded functions that can still bind a correct T so we can extract real instance from their arg. They forward calls to bounded API
233    fn drop_vtable(this: NonNull<TaskHeader>) {
234        let instance = this.as_ptr().cast::<AsyncTask<T>>();
235        unsafe { drop(ArcInternal::from_raw(instance)) };
236    }
237
238    fn clone_vtable(this: NonNull<TaskHeader>) {
239        let instance = this.as_ptr().cast::<AsyncTask<T>>();
240        unsafe { ArcInternal::increment_strong_count(instance) };
241    }
242
243    fn schedule_vtable(this: NonNull<TaskHeader>, task: TaskRef) {
244        let instance = this.as_ptr().cast::<AsyncTask<T>>();
245        unsafe { (*instance).schedule(task) }
246    }
247
248    fn poll_vtable(this: NonNull<TaskHeader>, ctx: &mut Context) -> TaskPollResult {
249        let instance = this.as_ptr().cast::<AsyncTask<T>>();
250        unsafe { (*instance).poll(ctx) }
251    }
252}
253
254///
255/// Creates static reference to VTable for specific `T` using `const promotion` of Rust
256///
257fn create_task_vtable<T: 'static>() -> &'static TaskVTable {
258    &TaskVTable {
259        drop: AsyncTask::<T>::drop_vtable,
260        clone: AsyncTask::<T>::clone_vtable,
261        schedule: AsyncTask::<T>::schedule_vtable,
262        poll: AsyncTask::<T>::poll_vtable,
263    }
264}

266///
267/// Task reference type that can track real `AsyncTask<T>` without knowing it's type. It make sure the ArcInternal<> holding a task has correct ref count
268///
269pub(crate) struct TaskRef {
270    header: NonNull<TaskHeader>,
271}
272
273unsafe impl Send for TaskRef {}
274
275impl TaskRef {
276    pub(crate) fn new<T>(arc_task: ArcInternal<AsyncTask<T>>) -> Self {
277        // Take raw pointer without any borrows
278        let val = NonNull::new(ArcInternal::as_ptr(&arc_task) as *mut TaskHeader).unwrap();
279        std::mem::forget(arc_task); // we took over ref count from arg into ourself
280        Self { header: val }
281    }
282
283    pub(crate) fn clone(&self) -> Self {
284        unsafe {
285            (self.header.as_ref().vtable.clone)(self.header);
286        }
287
288        Self { header: self.header }
289    }
290
291    ///
292    /// Release pointer and does not decrement ref count
293    ///
294    pub(crate) fn into_raw(this: TaskRef) -> *const TaskHeader {
295        let ptr = this.header.as_ptr();
296        std::mem::forget(this);
297        ptr
298    }
299
300    ///
301    /// Binds pointer and does not increment ref count. This call has to be paired with `into_raw` - same as ArcInternal<> calls.
302    ///
303    pub(crate) unsafe fn from_raw(ptr: *const TaskHeader) -> TaskRef {
304        Self {
305            header: NonNull::new(ptr as *mut TaskHeader).unwrap(),
306        }
307    }
308
309    ///
310    /// Proxy to `AsyncTask<T>::schedule`
311    ///
312    pub(crate) fn schedule(&self) {
313        unsafe {
314            (self.header.as_ref().vtable.schedule)(self.header, self.clone());
315        }
316    }
317
318    ///
319    /// Proxy to `AsyncTask<T>::poll`
320    ///
321    pub(crate) fn poll(&self, ctx: &mut Context) -> TaskPollResult {
322        unsafe { (self.header.as_ref().vtable.poll)(self.header, ctx) }
323    }
324}
325
326impl Drop for TaskRef {
327    fn drop(&mut self) {
328        unsafe {
329            (self.header.as_ref().vtable.drop)(self.header);
330        }
331    }
332}
